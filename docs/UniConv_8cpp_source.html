39UniConv::IConvResult UniConv::ConvertEncoding(const std::string& input, const char* fromEncoding, const char* toEncoding) {
40 IConvResult result;
41
42 if (input.empty()) {
43 result.error_code = 0;
44 result.conv_result_str = "";
45 return result;
46 }
47
48 iconv_t cd = iconv_open(toEncoding, fromEncoding);
49 if (cd == (iconv_t)-1) {
50 result.error_code = errno;
51 result.error_msg = "Failed to open iconv: " + std::string(strerror(errno));
52 return result;
53 }
54 // Prepare input and output buffers
55 size_t inbytesleft = input.size();
56 size_t outbytesleft = inbytesleft * 4; // Large enough output buffer for worst case
57 std::string output(outbytesleft, '\0');
58
59 const char* inbuf = const_cast<char*>(input.data());
60 char* outbuf = &output[0];
61 char* outbuf_start = outbuf;
62
63 // Perform conversion
64 size_t ret = iconv(cd, &inbuf, &inbytesleft, &outbuf, &outbytesleft);
65
66 if (ret == (size_t)-1) {
67 result.error_code = errno;
68 result.error_msg = "iconv conversion failed: " + std::string(strerror(errno));
69 iconv_close(cd);
70 return result;
71 }
72
73 // Calculate actual output size
74 size_t converted_size = outbuf - outbuf_start;
75 output.resize(converted_size);
76
77 result.conv_result_str = std::move(output);
78 result.error_code = 0;
79
80 iconv_close(cd);
81 return result;
82}

857std::string UniConv::WideStringToLocale(const std::wstring& sInput) {
858#ifdef _WIN32
859 if (sInput.empty()) return "";
860
861 int bytes_needed = WideCharToMultiByte(CP_ACP, 0, sInput.c_str(), -1, nullptr, 0, nullptr, nullptr);
862 if (bytes_needed <= 0) return "";
863
864 std::string result(bytes_needed - 1, '\0');
865 WideCharToMultiByte(CP_ACP, 0, sInput.c_str(), -1, &result[0], bytes_needed, nullptr, nullptr);
866 return result;
867#else
868 // Linux
869 std::wstring_convert<std::codecvt_byname<wchar_t, char, std::mbstate_t>> converter(new std::codecvt_byname<wchar_t, char, std::mbstate_t>(""));
870 return converter.to_bytes(sInput);

975std::string UniConv::GetIconvErrorString(int err_code) {
976 static const std::unordered_map<int, std::string> errorMap = {
977 {EILSEQ, "Invalid multibyte sequence"},
978 {EINVAL, "Incomplete multibyte sequence"},
979 {E2BIG, "Output buffer too small"},
980 {EBADF, "Invalid conversion descriptor"}
981 };
982
983 auto it = errorMap.find(err_code);
984 if (it != errorMap.end()) {
985 return it->second;
986 } return "Unknown iconv error: " + std::to_string(err_code);
