203std::pair<std::string, std::string> DetectEncodingAndRemoveBOM(const std::string& data) {
204 if (data.empty()) {
205 return std::make_pair("UTF-8", data);
206 }
207 if (data.size() >= 3 &&
208 static_cast<unsigned char>(data[0]) == 0xEF &&
209 static_cast<unsigned char>(data[1]) == 0xBB &&
210 static_cast<unsigned char>(data[2]) == 0xBF) {
211 return std::make_pair("UTF-8", data.substr(3));
212 }
213
214 if (data.size() >= 2 &&
215 static_cast<unsigned char>(data[0]) == 0xFF &&
216 static_cast<unsigned char>(data[1]) == 0xFE) {
217 return std::make_pair("UTF-16LE", data.substr(2));
218 }
219
220 if (data.size() >= 2 &&
221 static_cast<unsigned char>(data[0]) == 0xFE &&
222 static_cast<unsigned char>(data[1]) == 0xFF) {
223 return std::make_pair("UTF-16BE", data.substr(2));
224 }
225
226 // No BOM found, return original data
227 return std::make_pair("", data);
228}
