233std::pair<std::string, std::string> DetectEncodingAndRemoveBOM(const std::string& data) {
234 if (data.empty()) {
235 return {"UTF-8", data};
236 }
237
238 // Detect UTF-8 BOM (EF BB BF)
239 if (data.size() >= 3 &&
240 static_cast<unsigned char>(data[0]) == 0xEF &&
241 static_cast<unsigned char>(data[1]) == 0xBB &&
242 static_cast<unsigned char>(data[2]) == 0xBF) {
243 return {"UTF-8", data.substr(3)};
244 }
245
246 // Detect UTF-16LE BOM (FF FE)
247 if (data.size() >= 2 &&
248 static_cast<unsigned char>(data[0]) == 0xFF &&
249 static_cast<unsigned char>(data[1]) == 0xFE) {
250 return {"UTF-16LE", data.substr(2)};
251 }
252
253 // Detect UTF-16BE BOM (FE FF)
254 if (data.size() >= 2 &&
255 static_cast<unsigned char>(data[0]) == 0xFE &&
256 static_cast<unsigned char>(data[1]) == 0xFF) {
257 return {"UTF-16BE", data.substr(2)};
258 }
259 // No BOM detected, return original data with empty encoding indicator
260 return {"", data};
261}
