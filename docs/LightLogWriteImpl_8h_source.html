386 void RunWriteThread() {
387 while (true) {
388 if (bHasLogLasting)
389 if (bLastingTmTags != (GetCurrsTimerTm().tm_hour > 12))
390 CreateLogsFile();
391 LightLogWrite_Info sLogMessageInf;
392 {
393 auto sLock = std::unique_lock<std::mutex>(pLogWriteMutex);
394 pWritedCondVar.wait(sLock, [this] {return !pLogWriteQueue.empty() || bIsStopLogging; });
395 if (bIsStopLogging && pLogWriteQueue.empty()) break; // Exit if stop flag is true and queue is empty
396 if (!pLogWriteQueue.empty()) {
397 sLogMessageInf = pLogWriteQueue.front();
398 pLogWriteQueue.pop();
399 std::cerr << "pop:" << Ucs4ConvertToUtf8(sLogMessageInf.sLogContentVal) << "\n";
400 }
401 }
402 if (!sLogMessageInf.sLogContentVal.empty() && pLogFileStream.is_open())
403 {
404 pLogFileStream << sLogMessageInf.sLogTagNameVal << L"-//>>>" << GetCurrentTimer() << " : " << sLogMessageInf.sLogContentVal << "\n";
405 }
406 }
407 pLogFileStream.close();
408 std::cerr << "Log write thread Exit\n";
409 }
